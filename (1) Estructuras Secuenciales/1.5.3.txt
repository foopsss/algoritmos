Ambiente

Proceso

	ganancia_pc := costo_pc * 0,12
	ganancia_imp := costo_imp * 0,7
	iva := (costo_pc + costo_imp + ganancia_pc + ganancia_imp) * 0,21
	total := costo_pc + costo_imp + ganancia_pc + ganancia_imp + iva
	
	Escribir("El precio total de las ventas es de: " total "pesos.")
