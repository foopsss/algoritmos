Acción 2.2.12 ES

Ambiente

	Población = Registro
		Clave = Registro
			Provincia: AN(100)
			Departamento: AN(100)
			Ciudad: AN(100)
		Fin Registro
		Barrio: AN(100)
		IDCasa: N(5)
		CantHab: N(3)
	Fin Registro

	ARCH: archivo de Población ordenado por Clave
	REG: Población

	PoblaciónSal = Registro
		Provincia: AN(100)
		Departamento: AN(100)
		CantHab: N(3)
	Fin Registro

	ARCH2: archivo de PoblaciónSal
	REG2: PoblaciónSal

	cant_hab_depto: entero
	resg_depto, resg_prov: alfanumérico
	
	Procedimiento Inicializar() ES
		resg_depto := REG.Clave.Departamento
		resg_prov := REG.Clave.Provincia
		cant_hab_depto := 0
	Fin Procedimiento
	
	Procedimiento CorteDepartamento() ES
		REG2.Provincia := resg_prov
		REG2.Departamento := resg_depto
		REG2.CantHab := cant_hab_depto
		GRABAR(ARCH2, REG2)
		cant_hab_depto := 0
		resg_depto := REG.Clave.Departamento
	Fin Procedimiento
	
	Procedimiento CorteProvincia() ES
		CorteDepartamento()
		resg_prov := REG.Clave.Provincia
	Fin Procedimiento

Proceso

	ABRIR E/ (ARCH)
	ABRIR S/ (ARCH2)
	LEER(ARCH, REG)

	Inicializar()

	Mientras NFDA(ARCH) Hacer
		Si resg_prov <> REG.Clave.Provincia Entonces
			CorteProvincia()
		Contrario
			Si resg_depto <> REG.Clave.Departamento Entonces
				CorteDepartamento()
			Fin Si
		Fin Si
		
		cant_hab_depto := REG.CantHab
		LEER(ARCH, REG)
	Fin Mientras
	
	CorteProvincia()

	CERRAR(ARCH)
	CERRAR(ARCH2)

Fin Acción
