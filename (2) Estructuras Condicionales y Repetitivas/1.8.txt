	edad := año_act - año_nac
	
	Si mes_act = mes_nac Entonces
		Si día_act < día_nac Entonces
			edad := (año_act - año_nac) - 1					
		Fin Si
	Contrario

			edad := (año_act - año_nac) - 1
		Fin Si
	Fin Si
